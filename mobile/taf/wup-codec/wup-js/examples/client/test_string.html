测试字符串编解码示例 

function TestString()
{
    //下面开始编码测试
    var os = new Taf.JceOutputStream();
    os.writeString(0, "腾讯武汉研发中心");

    //下面开始解码测试
    os.getBinBuffer().position = 0;
    var is = new Taf.JceInputStream(os.getBinBuffer());
    var sr = is.readString(0, true, "");

    console.log("STRING:", sr);        		
}
