测试Vector编解码示例  元素类型为结构体

function TestVectorStruct()
{
    var sa = new TestJCE.UserInfo();
    var sb = new TestJCE.UserInfo();
    sa.stBasic.sUserID = "AA";
    sa.sToken = "AATOKEN";

    sb.stBasic.sUserID = "BB";
    sb.sToken = "BBTOKEN";

    var vo = new Taf.Vector(new TestJCE.UserInfo());
    vo.val.push(sa);
    vo.val.push(sb);

    //开始编码
    var os = new Taf.JceOutputStream();
    os.writeVector(5, vo);
    Taf.Util.jcestream(os.getBinBuffer().buf);

    //开始解码
    os.getBinBuffer().position = 0;
    var is = new Taf.JceInputStream(os.getBinBuffer());

    var vi = is.readVector(5, true, new Taf.Vector(new TestJCE.UserInfo()));
    console.log(">>>:", vi.toString());
}
